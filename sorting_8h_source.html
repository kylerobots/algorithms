37  inline std::vector<double> insertionSort(const std::vector<double> & input) {
38  auto output = input;
39  for (size_t i = 1; i < output.size(); ++i) {
40  double key = output[i];
41  // Since i starts at 1, j never starts less than zero.
42  size_t j = i - 1;
43  while (j != size_t(-1) && output[j] > key) {
44  output[j + 1] = output[j];
45  j--;
46  }
47  output[j + 1] = key;
48  }
49  return output;
50  }

73  inline void maxHeapify(std::vector<double> & array, size_t heap_size, size_t current_index) {
74  size_t left = 2 * current_index + 1;
75  size_t right = 2 * current_index + 2;
76  size_t largest = current_index;
77  // Determine if the target value is smaller than its potential
78  // children, which violates the heap property.
79  if (left <= heap_size - 1 && array[left] > array[current_index]) {
80  largest = left;
81  }
82  if (right <= heap_size - 1 && array[right] > array[largest]) {
83  largest = right;
84  }
85  // If it is smaller, replace it with the largest child, which
86  // restores the heap property.
87  if (largest != current_index) {
88  double key = array[current_index];
89  array[current_index] = array[largest];
90  array[largest] = key;
91  // Continue down the tree to verify that the new location does
92  // not violate the heap property for one of its new children.
93  maxHeapify(array, heap_size, largest);
94  }
95  }

106  inline std::vector<double> buildMaxHeap(const std::vector<double> & input) {
107  auto result = input;
108  // The leaves are guaranteed to be in heap order, since they have
109  // no children. So skip past them for fixing the heap.
110  // Since the index is always 0 or positive, casting to size_t is the
111  // same as the floor operation.
112  size_t start_index = static_cast<size_t>((result.size() - 1) / 2.0);
113  for (size_t i = start_index; i != static_cast<size_t>(-1); --i) {
114  detail::maxHeapify(result, result.size(), i);
115  }
116  return result;
117  }

136  left.push_back(input[p + i]);

189  inline size_t quickPartition(std::vector<double> & input, size_t start_index, size_t end_index) {
190  double pivot_value = input[end_index];
191  // Track where the pivot should go
192  size_t pivot_index = start_index - 1;
193  for (size_t i = start_index; i < end_index; ++i) {
194  // Go through each element. If it is less then the pivot,
195  // it should be on the left, so increment where the pivot will
196  // ultimately be placed.
197  if (input[i] <= pivot_value) {
198  pivot_index++;
199  // Then, move the value to the last spot of the left items.
200  double key = input[pivot_index];
201  input[pivot_index] = input[i];
202  input[i] = key;
203  }
204  }
205  // Once everything is partitioned, put the pivot in the right spot.
206  double key = input[pivot_index + 1];
207  input[pivot_index + 1] = input[end_index];
208  input[end_index] = key;
209  return pivot_index + 1;
210  }

241  inline std::vector<unsigned int> radixCountSort(const std::vector<unsigned int> & input, unsigned int digit) {
242  auto output = input;
243  // Allocate space to count the occurrence of each digit. There is at most
244  // 10 numbers to consider.
245  std::vector<unsigned int> value_counts(10, 0);
246  // Count the occurrence of each digit.
247  for (auto && i : input) {
248  // The digit under consideration is found by:
249  // value = (number / 10^digit) % 10
250  // The parenthesis moves the target digit to the ones place. The modulus
251  // then determines the remainder to figure out what this digit is. Since
252  // we are looking for an integer anyways, don't worry about truncating
253  // from rounding.
254  unsigned int value = static_cast<unsigned int>((i / std::pow(10, digit))) % 10;
255  value_counts[value]++;
256  }
257  // Now go through and perform summations to account for multiple instances
258  // of the same value.
259  for (size_t i = 1; i < value_counts.size(); ++i) {
260  value_counts[i] += value_counts[i - 1];
261  }
262  // Use these counts to determine where each element should go in the output
263  // array.
264  for (size_t i = input.size() - 1; i < static_cast<size_t>(-1); --i) {
265  // Determine which value is currently under consideration.
266  unsigned int current_value = static_cast<unsigned int>((input[i] / std::pow(10, digit))) % 10;
267  // Find how many elements occur before it
268  auto index_position = value_counts[current_value] - 1;
269  // Use that number of items to place it in the appropriate spot in
270  // the sorted array.
271  output[index_position] = input[i];
272  // output[value_counts[input[i]]] = input[i];
273  // Decrement each count so items don't get overwritten.
274  value_counts[current_value] -= 1;
275  }
276  return output;
277  }

310  inline std::vector<double> heapSort(const std::vector<double> & input) {
311  // This won't work for empty arrays because of the comparisions, so
312  // just return if so.
313  if (input.size() < 1) {
314  return input;
315  }
316  // First, build the heap structure
317  auto output = detail::buildMaxHeap(input);
318  size_t heap_size = output.size();
319  // Since the heap is ordered with the max element in the first position,
320  // move that element to the last position and reapply the heap
321  // property. We don't need to do 0, since it will be in the
322  // right spot by default.
323  for (size_t i = heap_size - 1; i > 0; --i) {
324  double key = output[i];
325  output[i] = output[0];
326  output[0] = key;
327  heap_size--;
328  detail::maxHeapify(output, heap_size, 0);
329  }
330  return output;
331  }

403  inline std::vector<unsigned int> radixSort(const std::vector<unsigned int> & input) {
404  auto output = input;
405  // Watch for empty arrays, since they won't sort correctly.
406  if (input.size() == 0) {
407  return output;
408  }
409  // Determine the max number to determine which digit to iterate up to.
410  unsigned int max = 0;
411  for (auto && number : input) {
412  max = std::max(max, number);
413  }
414  // The floor of log10 determines the index, since we are starting from 0. Since
415  // log10(10) = 1, anything less (AKA 1 digit) will be 0. Then anytime a new digit
416  // is added to the number, the result of log10 will also increase one. Because of
417  // the floor, casting to int won't lose any data.
418  unsigned int max_digit = static_cast<unsigned int>(std::floor(std::log10(max)));
419  for (unsigned int i = 0; i <= max_digit; ++i) {
420  output = detail::radixCountSort(output, i);
421  }
422  return output;
423  }

440  inline std::vector<double> bucketSort(const std::vector<double> & input) {
441  // Create each bucket
442  std::vector<std::vector<double>> buckets(input.size(), std::vector<double>());
443  // Place each number into a bucket. Since numbers are on the range [0, 1),
444  // multiplying by the total count of numbers ensures the floor yields the index
445  // that number should be placed in. This is because the number of buckets is
446  // also based on the total count of numbers. For example, consider 4 numbers.
447  // The second bucket is the range [0.25, 0.5). So floor(4 * 0.3) = 1, which is
448  // the right index.
449  for (auto && i : input) {
450  // Check that the value is in the correct range.
451  if (i >= 1.0 || i < 0.0) {
452  throw std::out_of_range("Keys must be in the range [0, )");
453  }
454  // Floor ensures the result is an integer, so is safe to cast.
455  size_t index = static_cast<size_t>(std::floor(input.size() * i));
456  buckets[index].push_back(i);
457  }
458  // Now sort each bucket using insertion sort. Although any sort would work.
459  // As the results are returned, place them in the output array.
460  std::vector<double> output;
461  for (auto && bucket : buckets) {
462  auto sorted_bucket = insertionSort(bucket);
463  for (auto && value : sorted_bucket) {
464  output.push_back(value);
465  }
466  }
467  return output;
468  }
