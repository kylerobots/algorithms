32  inline std::tuple<size_t, size_t, double> findMaxCrossingArray(const std::vector<float> & input, size_t low, size_t mid, size_t high) {
33  double left_sum = std::numeric_limits<double>::lowest();
34  double sum = 0.0;
35  size_t max_left = mid;
36  // If low is zero, this will wrap around to the max value. Since i starts at the midpoint, it
37  // will always be less than size_t::max, so the only time the comparison will be true is if
38  // the end of the iteration has truly been reached.
39  for (size_t i = mid; i >= low && i != size_t(-1); --i) {
40  sum += input[i];
41  if (sum > left_sum) {
42  left_sum = sum;
43  max_left = i;
44  }
45  }
46  double right_sum = std::numeric_limits<double>::lowest();
47  sum = 0.0;
48  size_t max_right = mid;
49  for (size_t i = mid + 1; i <= high; ++i) {
50  sum += input[i];
51  if (sum > right_sum) {
52  right_sum = sum;
53  max_right = i;
54  }
55  }
56  return std::make_tuple(max_left, max_right, left_sum + right_sum);
57  }

73  inline std::tuple<size_t, size_t, double> findMaxSubarray(const std::vector<float> & input, size_t low, size_t high) {
74  // If the array is only one element, the subarray is trivial to computer.
75  if (high == low) {
76  return std::make_tuple(low, high, input[low]);
77  } else {
78  // Since indices are always positive, casting will drop the decimal, which is
79  // the equivalent of floor.
80  size_t mid = static_cast<size_t>((low + high) / 2.0);
81  size_t left_low;
82  size_t left_high;
83  size_t right_low;
84  size_t right_high;
85  size_t cross_low;
86  size_t cross_high;
87  double left_sum;
88  double right_sum;
89  double cross_sum;
90  // Determine the three options
91  std::tie(left_low, left_high, left_sum) = findMaxSubarray(input, low, mid);
92  std::tie(right_low, right_high, right_sum) = findMaxSubarray(input, mid + 1, high);
93  std::tie(cross_low, cross_high, cross_sum) = findMaxCrossingArray(input, low, mid, high);
94  // Compare to see which is actually the maximum array.
95  if (left_sum >= right_sum && left_sum >= cross_sum) {
96  return std::make_tuple(left_low, left_high, left_sum);
97  } else if (right_sum >= left_sum && right_sum >= cross_sum) {
98  return std::make_tuple(right_low, right_high, right_sum);
99  } else {
100  return std::make_tuple(cross_low, cross_high, cross_sum);
101  }
102  }
103  }

119 inline std::tuple<size_t, size_t, double> maxSubarray(const std::vector<float> & input) {
120  // Verify that there are at least two elements in order to even form a difference between values.
121  if (input.size() < 2) {
122  throw std::invalid_argument("Provided array must have at least two elements.");
123  }
124  // First, create an array of differences.
125  std::vector<float> differences;
126  for (size_t i = 1; i < input.size(); ++i) {
127  float difference = input[i] - input[i - 1];
128  differences.push_back(difference);
129  }
130  // Now, call the recursive routine to find the array.
131  auto result = detail::findMaxSubarray(differences, 0, differences.size() - 1);
132  // These are indices for the difference array. Create the corresponding indices
133  // for the actual array.
134  size_t low_index = std::get<0>(result);
135  size_t high_index = std::get<1>(result);
136  double sum = std::get<2>(result);
137  return std::make_tuple(low_index, high_index + 1, sum);
138 }
