37  inline double memoizedCutRod(unsigned int length, const std::map<unsigned int, double> & prices, std::vector<double> solutions) {
38  // Return a value if it has already been computed.
39  if (solutions[length] >= 0) {
40  return solutions[length];
41  }
42  double best_price;
43  // We know a length of 0 has a price of 0.0.
44  if (length == 0) {
45  best_price = 0.0;
46  } else {
47  best_price = std::numeric_limits<double>::lowest();
48  // Recursively calculate the optimal solution for this
49  // length by picking the first cut and computing the
50  // optimal solution for the remaining length.
51  for (unsigned int i = 1; i <= length; ++i) {
52  double candidate_price = prices.at(i) + memoizedCutRod(length - i, prices, solutions);
53  best_price = std::max(best_price, candidate_price);
54  }
55  }
56  // Store this price for future lookup.
57  solutions[length] = best_price;
58  return best_price;
59  }

74  inline double bestRodCutBottomUp(unsigned int length, std::map<unsigned int, double> prices) {
75  // Create the storage for the computed values.
76  std::vector<double> solutions(length + 1, 0.0);
77  solutions[0] = 0.0;
78  for (size_t i = 1; i <= length; ++i) {
79  double best_price = std::numeric_limits<double>::lowest();
80  // The solution is determined by the exploration of all possible
81  // first cuts plus the optimal solution of the remaining length.
82  // Since this is bottom up, the remaining length solutions have
83  // already been computed.
84  for (unsigned int j = 1; j <= i; ++j) {
85  // Prices[j] represents the price for the first cut. Solutions[i-j]
86  // is then the ideal price for the remaining length.
87  best_price = std::max(best_price, prices.at(j) + solutions[i - j]);
88  }
89  // Store the best price for this length for use by higher solutions.
90  solutions[i] = best_price;
91  }
92  return solutions[length];
93  }
